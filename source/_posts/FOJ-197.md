```cpp
#include<iostream>
#include<string>
using namespace std;
int main(){
  int t;
  cin>>t;
  while(t--){
    string s;
    int L=0, O=0, V=0, E=0, ans=0;
    cin>>s;
    for(int i=0; i<s.size(); i++){
      if(s[i]=='L'){
        L++;
      }
      else if(s[i]=='O'){
        if(O<L){
          O++;
        }
      }
      else if(s[i]=='V'){
        if(V<O){
          V++;
        }
      }
      else{
        if(E<V){
          E++;
        }
      }
    }
    cout<<E<<"\n";
  }
}
```
